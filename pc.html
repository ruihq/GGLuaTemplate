Welcome to the PC Version

This is the PC version of the RHQ Script Template.

Starting hook

The starting hook is used to initialize the script and display a loading message.


gg.toast("Starting hook...");
gg.sleep(2250);

Welcome Phase

The welcome phase displays the initial alerts and loading progress.


gg.setVisible(false);
gg.alert(getWelcomeText());
gg.clearResults();
gg.toast(getLoadingProgress());
gg.sleep(500);

function getWelcomeText() {
    var languageSelect = document.getElementById("language");
    var language = languageSelect.options[languageSelect.selectedIndex].value;
    if (language === "tl") {
        return "Maligayang pagdating sa RHQ Script Template";
    } else {
        return "Welcome to the RHQ Script Template";
    }
}

function getLoadingProgress() {
    var languageSelect = document.getElementById("language");
    var language = languageSelect.options[languageSelect.selectedIndex].value;
    if (language === "tl") {
        return "\n■";
    } else {
        return "\n■";
    }
}

Menu

The menu displays the available options and triggers the corresponding functions based on the user's choice.


menu = gg.choice({
    "Option 1",
    "Option 2",
    "Option 3",
    "Option 4"
}, nil, getMenuTitle());

if menu == 1 then option1() end;
if menu == 2 then option2() end;
if menu == 3 then option3() end;
if menu == 4 then option4() end;

function getMenuTitle() {
    var languageSelect = document.getElementById("language");
    var language = languageSelect.options[languageSelect.selectedIndex].value;
    if (language === "tl") {
        return "Home Menu - RHQ Script Template";
    } else {
        return "Home Menu - RHQ Script Template";
    }
}
